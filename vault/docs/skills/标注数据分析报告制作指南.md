### 2. JSON 数据展平

VLM 输出的 JSON 容易出现 schema 问题，需要防御性解析：

```python
def safe_get(obj, key, default=None):
    """安全获取字典值，处理非字典类型"""
    if isinstance(obj, dict):
        return obj.get(key, default)
    return default

def flatten_annotation(data: dict) -> dict | None:
    """将嵌套 JSON 展平为单层字典，格式不正确返回 None"""
    if not isinstance(data, dict):
        return None
    flat = {}
    # 逐字段安全提取...
    return flat

# 批量处理，记录解析错误
rows, parse_errors = [], []
for row in results:
    try:
        data = json.loads(row["value_json"])
        flat = flatten_annotation(data)
        if flat is None:
            parse_errors.append({"id": str(row["id"]), "error": "flatten failed"})
            continue
        rows.append(flat)
    except (json.JSONDecodeError, TypeError) as e:
        parse_errors.append({"id": str(row["id"]), "error": str(e)})
```

### 3. 标签验证与过滤

根据 prompt schema 定义有效标签集合，过滤掉 VLM 生成的无效选项：

```python
# 定义有效标签集合
VALID_TAGS = {"tag1", "tag2", "tag3"}

TAG_MEANINGS = {

}

def filter_valid_counter(counter: Counter, valid_set: set) -> tuple[Counter, Counter]:
    """分离有效和无效标签"""
    valid = Counter({k: v for k, v in counter.items() if k in valid_set})
    invalid = Counter({k: v for k, v in counter.items() if k not in valid_set})
    return valid, invalid
```

**饼图避免标签重叠**（数据分布极不均匀时）：
```python
def plot_pie_chart(counter: Counter, title: str, output_path: Path):
    sorted_items = counter.most_common()
    labels, values = zip(*sorted_items)
    total = sum(values)

    fig, ax = plt.subplots(figsize=(12, 8))

    # 小切片只显示百分比，大切片显示完整信息
    def autopct_func(pct):
        return f"{pct:.1f}%" if pct >= 3 else ""

    # 小切片稍微分离以便区分
    explode = [0.02 if v / total < 0.05 else 0 for v in values]

    wedges, _, autotexts = ax.pie(
        values,
        labels=None,  # 不在饼图上显示标签，避免重叠
        autopct=autopct_func,
        explode=explode,
        startangle=90,
    )

    # 使用图例显示完整信息
    legend_labels = [f"{label} ({count:,}, {count/total*100:.1f}%)" for label, count in sorted_items]
    ax.legend(wedges, legend_labels, loc="center left", bbox_to_anchor=(1.0, 0.5))

    plt.tight_layout()
    plt.savefig(output_path, dpi=150, bbox_inches="tight")
```
